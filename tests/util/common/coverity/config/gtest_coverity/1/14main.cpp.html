81   	static int
82   	parse_dpdk_config(const char *q_arg)
83   	{
84   		char s[256];
85   		const char *p, *p0 = q_arg;
86   		char *end;
87   		enum fieldnames {
88   			FLD_NUM_QUEUES = 0,
89   			_NUM_FLD
90   		};
91   		unsigned long int_fld[_NUM_FLD];
92   		char *str_fld[_NUM_FLD];
93   		int i;
94   		unsigned size;
95   	
96   		while ((p = strchr(p0,'(')) != NULL) {
97   			++p;
98   			if((p0 = strchr(p,')')) == NULL)
99   				return -1;
100  	
101  			size = p0 - p;
102  			if(size >= sizeof(s))
103  				return -1;
104  	
105  			snprintf(s, sizeof(s), "%.*s", size, p);
106  			if (rte_strsplit(s, sizeof(s), str_fld, _NUM_FLD, ',') != _NUM_FLD)
107  				return -1;
108  			for (i = 0; i < _NUM_FLD; i++){
109  				errno = 0;
110  				int_fld[i] = strtoul(str_fld[i], &end, 0);
111  				if (errno != 0 || end == str_fld[i])
112  					return -1;
113  			}
114  	
115  			config_max_queues = (unsigned int)int_fld[FLD_NUM_QUEUES];
116  			if(config_max_queues > MAX_NUM_QUEUES)
117  			{
118  				printf("Invalid number of Queues\n");
119  				return -1;
120  			}
121  		}
122  		return 0;
123  	}

179  	static unsigned int num_trailing_blanks(char *word)
180  	{
181  	    unsigned int i = 0;
182  	    size_t slen = strlen(word);
183  	
184  	    if (!slen) return 0;
185  	    while (isspace(word[slen - i - 1])) {
186  	        i++;
187  	    }
188  	
189  	    return i;
190  	}

192  	static char * strip_blanks(char *word, size_t *blanks)
193  	{
194  	    char *p = word;
195  	    unsigned int i = 0;
196  	
197  	    while (isblank(p[0])) {
198  	        p++;
199  	        i++;
200  	    }
201  	    if (blanks)
202  	        *blanks = i;
203  	
204  	    return p;
205  	}

207  	static char * strip_comments(char *word)
208  	{
209  	    size_t numblanks;
210  	    char *p = strip_blanks(word, &numblanks);
211  	
212  	    if (p[0] == '#')
213  	        return NULL;
214  	    else
215  	        p = strtok(word, "#");
216  	
217  	    return p;
218  	}

233  	static int arg_read_int_array(char *s, unsigned int *v, unsigned int max_arr_size)
234  	{
235  	    size_t slen = strlen(s);
236  	    unsigned int trail_blanks = num_trailing_blanks(s);
237  	    char *str = (char *)malloc(slen - trail_blanks + 1);
238  	    char *elem;
239  	    int cnt = 0;
240  	    int ret;
241  	
242  	    memset(str, '\0', slen + 1);
243  	    strncpy(str, s + 1, slen - trail_blanks - 2);
244  	    str[slen] = '\0';
245  	
246  	    elem = strtok(str, " ,");/* space or comma separated */
247  	    while (elem != NULL) {
248  	        ret = arg_read_int(elem, &v[cnt]);
249  	        if (ret < 0) {
250  	            printf("ERROR: Invalid array element %sin %s\n", elem, s);
251  	            exit(0);
252  	        }
253  	        cnt++;
254  	        elem = strtok(NULL, " ,");
255  	        if (cnt > (int)max_arr_size) { /* to avoid out of bounds */
256  	            printf("ERROR: More than expected number of elements in %s - expected = %u\n",
257  	                   str, max_arr_size);
258  	            exit(0);
259  	        }
260  	    }
261  	    free(str);
262  	
263  	    return cnt;
264  	}

312  	static int get_array_len(char *s)
313  	{
314  	    int i, len = 0;
315  	
316  	    if (strlen(s) < 2)
317  	        return -EINVAL;
318  	    if ((s[0] != '(') && (s[strlen(s) - 1] != ')'))
319  	        return -EINVAL;
320  	    if ((s[0] == '(') && (s[1] == ')'))
321  	        return 0;
322  	    for (i = 0; i < (int)strlen(s); i++) {
323  	        if ((s[i] == ' ') || (s[i] == ',')) /* space or comma separated */
324  	                len++;
325  	        if (s[i] == ')')
326  	            break;
327  	    }
328  	
329  	    return (len + 1);
330  	}

332  	static void parse_config_file(const char *cfg_fname)
333  	{
334  	    char *realbuf;
335  	    size_t linelen = 0;
336  	    size_t numblanks;
337  	    char *config, *value;
338  	
339  	    std::ifstream infile;
340  	    auto filestate = infile.exceptions();
341  	    infile.exceptions(std::ifstream::failbit | std::ifstream::badbit);
342  	
343  	    try {
344  	        infile.open(cfg_fname, std::ios_base::in);
345  	    }
346  	    catch (std::ifstream::failure& e) {
347  	        std::cerr << "Exception happened: " << e.what() << std::endl;
348  	        std::cout << "libc error #" << ": " << ::strerror(errno) << std::endl;
349  	        std::cout << "handled" << std::endl;
350  	        exit(EXIT_FAILURE);
351  	    }
352  	
353  	    infile.exceptions(filestate);
354  	
355  	    for (std::string line; std::getline(infile, line); ) {
356  	        char *linebuf = new char[line.length() + 1];
357  	        std::unique_ptr<char> smart_ptr(linebuf);
358  	
359  	        strcpy(linebuf, line.c_str());
360  	        linelen = line.length();
361  	
362  	        linebuf = strip_comments(linebuf);
363  	        if (!linebuf)
364  	            continue;
365  	
366  	        realbuf = strip_blanks(linebuf, &numblanks);
367  	        linelen -= numblanks;
368  	        if (0 == linelen) {
369  	            continue;
370  	        }
371  	        if (strlen(realbuf) == 0) {
372  	            continue;
373  	        }
374  	
375  	        config = strtok(realbuf, "=");
376  	        value = strtok(NULL, "=");
377  	        if (!strncmp(config, "master_pf", 8)) {
378  	            if (arg_read_int(value, &env.sys_cfg.master_pf)) {
379  	                printf("Error: Invalid master_pf:%s\n", value);
380  	                exit(1);
381  	            }
382  	        } else if (!strncmp(config, "qmaxvfs", 7)) {
383  	                if (arg_read_int(value, &env.sys_cfg.qmaxvfs)) {
384  	                    printf("Error: Invalid qmaxvfs:%s\n", value);
385  	                    exit(1);
386  	                }
387  	        } else if (!strncmp(config, "pf_qmax_cfg_en", 14)) {
388  	            int arr_len = get_array_len(value);
389  	            int ret;
390  	
391  	            if (arr_len < 0) {
392  	                printf("ERROR: Invalid pf_qmax_cfg_en - %s\n", value);
393  	                exit(1);
394  	            }
395  	            env.sys_cfg.num_pf = arr_len;
396  	            if (arr_len) {
397  	                env.sys_cfg.pf_qmax_cfg_en = (unsigned int *)calloc(arr_len, sizeof(unsigned int));
398  	                ret = arg_read_int_array(value, env.sys_cfg.pf_qmax_cfg_en, arr_len);
399  	                if (ret <= 0) {
400  	                    printf("Error: Invalid pf_qmax_cfg_en:%s\n", value);
401  	                    exit(1);
402  	                }
403  	            }
404  	        } else if (!strncmp(config, "pf_qmax", 7)) {
405  	            int arr_len = get_array_len(value);
406  	            int ret;
407  	
408  	            if ((arr_len < 0) || (arr_len != (int)env.sys_cfg.num_pf)) {
409  	                printf("ERROR: Invalid number of entries in pf_qmax - %s\n", value);
410  	                exit(1);
411  	            }
412  	            if (arr_len) {
413  	                env.sys_cfg.pf_qmax_lst = (unsigned int *)calloc(arr_len, sizeof(unsigned int));
414  	                ret = arg_read_int_array(value, env.sys_cfg.pf_qmax_lst, arr_len);
415  	                if (ret <= 0) {
416  	                    printf("Error: Invalid pf_qmax_lst:%s\n", value);
417  	                    exit(1);
418  	                }
419  	            }
420  	        } else if (!strncmp(config, "pf_nvf", 6)) {
421  	            int arr_len = get_array_len(value);
422  	            int ret, i;
423  	
424  	            if ((arr_len < 0) || (arr_len != (int)env.sys_cfg.num_pf)) {
425  	                printf("ERROR: Invalid number of entries in pf_nvf_lst - %s\n", value);
426  	                exit(1);
427  	            }
428  	            if (arr_len) {
429  	                env.sys_cfg.pf_nvf_lst = (unsigned int *)calloc(arr_len, sizeof(unsigned int));
430  	                ret = arg_read_int_array(value, env.sys_cfg.pf_nvf_lst, arr_len);
431  	                if (ret <= 0) {
432  	                    printf("Error: Invalid pf_nvf_lst:%s\n", value);
433  	                    exit(1);
434  	                }
435  	            }
436  	            env.sys_cfg.vf_pf_qmax_lst = (unsigned int **)calloc(arr_len, sizeof(unsigned int *));
437  	            for (i = 0; i < arr_len; i++)
438  	                env.sys_cfg.vf_pf_qmax_lst[i] = (unsigned int *)calloc(env.sys_cfg.pf_nvf_lst[i], sizeof(unsigned int));
439  	        } else if (!strncmp(config, "vf_pf", 5)) {
440  	            size_t slen = strlen(config);
441  	            char *str;
442  	            int arr_len;
443  	            int ret;
444  	            int pf = 0;
445  	
446  	            if (strncmp(config+slen-5, "_qmax", 5)) {
447  	                printf("ERROR: unknown config param %s\n", config);
448  	                exit(1);
449  	            }
450  	            if (!env.sys_cfg.pf_nvf_lst) {
451  	                printf("ERROR: set pf_nvf_lst before vf_pf*_qmax\n");
452  	                exit(1);
453  	            }
454  	            str = strtok(config+5, "_");
455  	            if (str != NULL) {
456  	                int r = arg_read_int(str, (unsigned int *)&pf);
457  	
458  	                if (r < 0) {
459  	                    printf("ERROR: Invalid PF number %s in %s\n", str, config);
460  	                    exit(1);
461  	                }
462  	                if (pf > (int)env.sys_cfg.num_pf) {
463  	                    printf("%d PF number does not exist. Invalid %s config\n", pf, config);
464  	                    exit(1);
465  	                }
466  	
467  	            }
468  	            arr_len = get_array_len(value);
469  	            if ((arr_len < 0) || (arr_len > (int)env.sys_cfg.pf_nvf_lst[pf])) {
470  	                printf("ERROR: Invalid number of entries in vf_pf%d_qmax - %s\n", pf, value);
471  	                exit(1);
472  	            }
473  	            if (arr_len) {
474  	                ret = arg_read_int_array(value, env.sys_cfg.vf_pf_qmax_lst[pf], arr_len);
475  	                if (ret <= 0) {
476  	                    printf("Error: Invalid vf_pf%d_qmax:%s\n", pf, value);
477  	                    exit(1);
478  	                }
479  	            }
480  	        } else if (!strncmp(config, "pci_bus", 7)) {
481  	            int arr_len = get_array_len(value);
482  	            int ret;
483  	
484  	            if (arr_len < 0) {
485  	                printf("ERROR: Invalid pci_bus - %s\n", value);
486  	                exit(1);
487  	            }
488  	            env.sys_cfg.num_cards = arr_len;
489  	            if (arr_len) {
490  	                env.sys_cfg.pci_bus = (unsigned int *)calloc(arr_len, sizeof(unsigned int));
491  	                ret = arg_read_hex_array(value, env.sys_cfg.pci_bus, arr_len);
492  	                if (ret <= 0) {
493  	                    printf("Error: Invalid pci_bus:%s\n", value);
494  	                    exit(1);
495  	                }
496  	            }
497  	        } else if (!strncmp(config, "pci_dev", 7)) {
498  	            int arr_len = get_array_len(value);
499  	            int ret;
500  	
501  	            if (arr_len < 0) {
502  	                printf("ERROR: Invalid pci_dev - %s\n", value);
503  	                exit(1);
504  	            }
505  	            if (arr_len) {
506  	                env.sys_cfg.pci_dev = (unsigned int *)calloc(arr_len, sizeof(unsigned int));
507  	                ret = arg_read_int_array(value, env.sys_cfg.pci_dev, arr_len);
508  	                if (ret <= 0) {
509  	                    printf("Error: Invalid pci_dev:%s\n", value);
510  	                    exit(1);
511  	                }
512  	            }
513  	        } else if(!strncmp(config, "multiple_apps", 7)) {
514  				if (arg_read_int(value, &env.sys_cfg.multiple_apps)) {
515  	                printf("Error: Invalid master_pf:%s\n", value);
516  	                exit(1);
517  	            }
518  			}
519  	        else {
520  	            printf("WARNING: Unknown parameter '%s'\n", config);
521  	        }
522  	    }
523  	
524  	    infile.close();
525  	}
