8    	int do_recv_mm(int portid, void *buffer, int queueid, int ld_size, int tot_num_desc, size_t buf_size, uint16_t rx_burst_size)
9    	{
10   		struct rte_mbuf *pkts[NUM_RX_PKTS] = { NULL };
11   		int nb_rx = 0, i = 0, num_pkts;
12   		int tdesc;
13   	
14   		if(tot_num_desc == 0) {
15   				printf("Error: tot_num_desc : invalid value\n");
16   				return 0;
17   		}
18   	
19   		//printf("recv start: num-desc: %d, with data-len: %d, last-desc-size:%d\n", tot_num_desc, buf_size, ld_size);
20   		tdesc = tot_num_desc;
21   		if (ld_size)
22   			tdesc--;
23   	
24   		while (tdesc) {
25   			if (tdesc > rx_burst_size)
26   				num_pkts = rx_burst_size;
27   			else
28   				num_pkts = tdesc;
29   	
30   			/* try to receive RX_BURST_SZ packets */
31   			nb_rx = rte_eth_rx_burst(portid, queueid, pkts, num_pkts);
32   			if (nb_rx == 0) {
33   				printf("Error: dma_from_device failed to receive packets\n");
34   				return 0;
35   			}
36   	
37   			for (i = 0; i < nb_rx; i++) {
38   				struct rte_mbuf *mb = pkts[i];
39   				memcpy(buffer, rte_pktmbuf_mtod(mb, void*), buf_size);
40   				buffer = static_cast<char*>(buffer) + buf_size;
41   				rte_pktmbuf_free(mb);
42   			}
43   	
44   			tdesc = tdesc - nb_rx;
45   		}
46   	
47   		if (ld_size) {
48   			struct rte_mbuf *mb;
49   			nb_rx = rte_eth_rx_burst(portid, queueid, pkts, 1);
50   			mb = pkts[0];
51   			if(nb_rx != 0) {
52   				memcpy(buffer, rte_pktmbuf_mtod(mb, void*), ld_size);
53   				buffer = static_cast<char*>(buffer) + ld_size;
54   				rte_pktmbuf_free(mb);
55   			}
56   		}
57   	
58   		//printf("\nDMA received number of packets: %d, on queue-id:%d\n", nb_rx, queueid);
59   		return 0;
60   	}

62   	int do_recv_st(int portid, void *buffer, int queueid, int input_size)
63   	{
64   		struct rte_mbuf *pkts[NUM_RX_PKTS] = { NULL };
65   		int nb_rx = 0, i, ret = 0,tmp = 0, num_pkts, nb_pkts, num_pkts_recv = 0, bytes	= 0;
66   		int reg_val;
67   		int user_bar_idx, dma_bar_idx, bypass_bar_idx, en_imm;
68   		struct rte_mbuf *nxtmb;
69   		unsigned int max_rx_retry;
70   	
71   	
72   	#ifdef DUMP_MEMPOOL_USAGE_STATS
73   		char	mem_pool[RTE_MEMPOOL_NAMESIZE];
74   		int q_dir = 1;
75   	
76   		snprintf(mem_pool, RTE_MEMPOOL_NAMESIZE, MBUF_POOL_NAME_PORT_QUEUE_DIR, portid, queueid, q_dir);
77   		struct rte_mempool *mp;
78   	
79   		/* get the mempool from which to acquire buffers */
80   		if ((mp = rte_mempool_lookup(mem_pool)) == NULL)
81   			throw std::runtime_error("Could not find mempool");
82   	#endif //DUMP_MEMPOOL_USAGE_STATS
83   		ret = rte_pmd_qdma_get_bar_details(portid, &(dma_bar_idx),
84   				&(user_bar_idx), &(bypass_bar_idx));
85   		if (ret < 0)
86   			throw std::runtime_error("rte_pmd_qdma_get_bar_details failed");
87   		ret = rte_pmd_qdma_get_immediate_data_state(portid, queueid, &(en_imm));
88   		if (ret < 0)
89   			throw std::runtime_error("rte_pmd_qdma_get_immediate_data_state failed");
90   	
91   		/* As per  hardware design a single writeback will point to atmost 7 descriptors.
92   		 * So If the size of the buffer in descriptor is 4KB, then a single writeback which
93   		 * corresponds a packet can give you atmost 28KB data.
94   		 *
95   		 * As per this when testing sizes beyond 28KB, one needs to split it up in chunks of 28KB,
96   		 * example : to test 56KB data size, set 28KB as packet length in USER BAR 0x04 register
97   		 * and no of packets as 2 in user BAR 0x20 register this would give you writebacks or packets,
98   		 * which needs to be combined as one in application.
99   		*/
100  	
101  		/* Calculate the number of packets to receive and programming the user bar */
102  		if(input_size == 0) /* zerobyte support uses one descriptor */
103  			num_pkts = 1;
104  		else
105  			num_pkts = qdma_pci_read_reg(&rte_eth_devices[portid], user_bar_idx, C2H_PACKET_COUNT_REG);
106  	/*	else if(input_size % MAX_WRITEBACK_SIZE != 0)
107  			num_pkts = input_size/MAX_WRITEBACK_SIZE + 1;
108  		else
109  			num_pkts = input_size/MAX_WRITEBACK_SIZE;*/
110  		reg_val = qdma_pci_read_reg(&rte_eth_devices[portid], user_bar_idx, C2H_CONTROL_REG);
111  		//reg_val &= C2H_CONTROL_REG_MASK;
112  		if (reg_val & ST_LOOPBACK_EN){
113  			do{
114  				nb_rx = rte_eth_rx_burst(portid, queueid, pkts, NUM_RX_PKTS);
115  				for (i = 0; i < nb_rx; i++) {
116  					struct rte_mbuf *mb = pkts[i];
117  					ret = 0;
118  					while (mb != NULL){
119  						memcpy(buffer, rte_pktmbuf_mtod(mb, void*), rte_pktmbuf_data_len(mb));
120  						ret += rte_pktmbuf_data_len(mb);
121  						buffer = static_cast<char*>(buffer) + rte_pktmbuf_data_len(mb);
122  						nxtmb = mb->next;
123  						mb = nxtmb;
124  					}
125  					mb = pkts[i];
126  					rte_pktmbuf_free(mb);
127  					//printf("recv count: %d, with data-len: %d\n", i, ret);
128  					bytes += ret;
129  				}
130  			}while(bytes < input_size);
131  			return bytes;
132  		}
133  	
134  		while (num_pkts) {
135  			if (num_pkts > NUM_RX_PKTS)
136  				nb_pkts = NUM_RX_PKTS;
137  			else
138  				nb_pkts = num_pkts;
139  	
140  			num_pkts_recv = nb_pkts;
141  			max_rx_retry = RX_TX_MAX_RETRY;
142  	
143  			if(en_imm) { /* Immediate data Enabled*/
144  				/* payload received is zero for the immediate data case. Therefore , no need
145  				 * to call the rx_burst function again in this case and set the num_pkts to nb_rx
146  				 * which is always Zero.
147  				 */
(2) Event returned_value: Assigning value from "rte_eth_rx_burst(portid, queueid, pkts, nb_pkts)" to "nb_rx" here, but that stored value is overwritten before it can be used.
Also see events: [value_overwrite]
148  				nb_rx = rte_eth_rx_burst(portid, queueid, pkts, nb_pkts);
149  				num_pkts = 0;
150  			}
151  			else {
152  				/* try to receive RX_BURST_SZ packets */
153
(1) Event value_overwrite: Overwriting previous write to "nb_rx" with value from "rte_eth_rx_burst(portid, queueid, pkts, nb_pkts)".
Also see events: [returned_value]
154  				nb_rx = rte_eth_rx_burst(portid, queueid, pkts, nb_pkts);
155  				tmp = nb_rx;
156  				if(input_size == 0)
157  					break;
158  				while((nb_rx < nb_pkts) && max_rx_retry) {
159  					/*printf("Couldn't receive all the packets:	 Expected = %d Received = %d. \n"
160  					"Calling rte_eth_rx_burst again\n", nb_pkts , nb_rx);*/
161  					usleep(1);
162  					nb_pkts -= nb_rx;
163  					nb_rx = rte_eth_rx_burst(portid, queueid, &pkts[tmp], nb_pkts);
164  					tmp += nb_rx;
165  					max_rx_retry--;
166  				}
167  	
168  				num_pkts_recv = tmp;
169  				if((max_rx_retry == 0) && (tmp == 0)) {
170  						printf("ERROR: rte_eth_rx_burst failed\n");
171  						break;
172  				}
173  	
174  	#ifdef DUMP_MEMPOOL_USAGE_STATS
175  				printf("%s(): %d: queue id = %d, mbuf_avail_count = %d, mbuf_in_use_count = %d\n",
176  					__func__, __LINE__, queueid, rte_mempool_avail_count(mp), rte_mempool_in_use_count(mp));
177  	#endif //DUMP_MEMPOOL_USAGE_STATS
178  				for (i = 0; i < num_pkts_recv; i++) {
179  					struct rte_mbuf *mb = pkts[i];
180  					while (mb != NULL){
181  						memcpy(buffer, rte_pktmbuf_mtod(mb, void*), rte_pktmbuf_data_len(mb));
182  						ret += rte_pktmbuf_data_len(mb);
183  						buffer = static_cast<char*>(buffer) + rte_pktmbuf_data_len(mb);
184  						nxtmb = mb->next;
185  						mb = nxtmb;
186  					}
187  	
188  					mb = pkts[i];
189  					rte_pktmbuf_free(mb);
190  					//printf("recv count: %d, with data-len: %d\n", i, ret);
191  					ret = 0;
192  				}
193  	#ifdef DUMP_MEMPOOL_USAGE_STATS
194  				printf("%s(): %d: queue id = %d, mbuf_avail_count = %d, mbuf_in_use_count = %d, num_pkts_recv = %d\n",
195  					__func__, __LINE__, queueid, rte_mempool_avail_count(mp), rte_mempool_in_use_count(mp), num_pkts_recv);
196  	#endif //DUMP_MEMPOOL_USAGE_STATS
197  				num_pkts = num_pkts - num_pkts_recv;
198  			}
199  		}
200  	
201  		//printf("\nDMA received number of packets: %d, on queue-id:%d\n", num_pkts, queueid);
202  	
203  		return 0;
204  	}

207  	int do_xmit(int portid, void * buffer, int queueid, int ld_size, int tot_num_desc, int zbyte, size_t buf_size, uint16_t tx_burst_size)
208  	{
209  		struct rte_mempool *mp;
210  		struct rte_mbuf *mb[NUM_RX_PKTS] = { NULL };
211  		int nb_tx = 0, i = 0, tdesc, num_pkts = 0, tmp = 0, nb_pkts = 0;
212  		uint32_t max_tx_retry;
213  		char	mem_pool[RTE_MEMPOOL_NAMESIZE];
214  		int	q_dir = 0;
215  		int loop_cnt = 5;
216  		uint16_t burst_size = tx_burst_size;
217  	
218  		snprintf(mem_pool, RTE_MEMPOOL_NAMESIZE, MBUF_POOL_NAME_PORT_QUEUE_DIR, portid, queueid, q_dir);
219  	
220  		while (((mp = rte_mempool_lookup(mem_pool)) == NULL) && loop_cnt >= 0)
221  		{
222  			usleep(100);
223  			loop_cnt--;
224  		}
225  	
226  		if (mp == NULL) {
227  			printf("do_xmit : Could not find mempool with name: %s\n", mem_pool);
228  			throw std::runtime_error("do_xmit : Could not find mempool with name:\n");
229  		}
230  		tdesc = tot_num_desc;
231  		if (ld_size)
232  			tdesc --;
233  	
234  		while (tdesc) {
235  			if (tdesc > burst_size)
236  				num_pkts = burst_size;
237  			else
238  				num_pkts = tdesc;
239  	
240  			max_tx_retry = RX_TX_MAX_RETRY;
241  	#ifdef DUMP_MEMPOOL_USAGE_STATS
242  			printf("%s(): %d: queue id %d, mbuf_avail_count = %d, mbuf_in_use_count = %d\n",
243  				__func__, __LINE__, queueid, rte_mempool_avail_count(mp), rte_mempool_in_use_count(mp));
244  	#endif //DUMP_MEMPOOL_USAGE_STATS
245  			for (i=0; i<num_pkts; i++) {
246  				mb[i] = rte_pktmbuf_alloc(mp);
247  				if (mb[i] == NULL) {
248  						throw std::runtime_error(" #####Cannot allocate mbuf packet ");
249  				}
250  				if (!zbyte){
251  					memcpy(rte_pktmbuf_mtod(mb[i], void *), buffer , buf_size);
252  					buffer = static_cast<char*>(buffer) + buf_size;
253  				}
254  				mb[i]->nb_segs = 1;
255  				mb[i]->next = NULL;
256  				rte_pktmbuf_data_len(mb[i]) = (!zbyte) ? (uint16_t)buf_size : 0;
257  				rte_pktmbuf_pkt_len(mb[i])	= (!zbyte) ? (uint16_t)buf_size : 0;
258  			}
259  	
260  	#ifdef DUMP_MEMPOOL_USAGE_STATS
261  			printf("%s(): %d: queue id %d, mbuf_avail_count = %d, mbuf_in_use_count = %d, num_pkts_tx = %d\n",
262  				__func__, __LINE__, queueid, rte_mempool_avail_count(mp), rte_mempool_in_use_count(mp), num_pkts);
263  	#endif //DUMP_MEMPOOL_USAGE_STATS
264  	
265  			/* try to transmit TX_BURST_SZ packets */
266  			nb_pkts = num_pkts;
267  			nb_tx = rte_eth_tx_burst(portid, queueid, mb, num_pkts);
268  			tmp = nb_tx;
269  			while((nb_tx < num_pkts) && max_tx_retry) {
270  				/*printf("Couldn't transmit all the packets: queueid=%d	 Expected = %d Transmitted = %d. \n"
271  						"Calling rte_eth_tx_burst again\n",queueid, num_pkts , nb_tx);*/
272  				usleep(1);
273  				num_pkts -= nb_tx;
274  				nb_tx = rte_eth_tx_burst(portid, queueid, &mb[tmp], num_pkts);
275  				tmp += nb_tx;
276  				max_tx_retry--;
277  			}
278  			if((max_tx_retry == 0)) {
279  				for (i = tmp; i < nb_pkts; i++)
280  					rte_pktmbuf_free(mb[i]);
281  				if (tmp == 0) {
282  					printf("ERROR: rte_eth_tx_burst failed \n");
283  					break;
284  				}
285  			}
286  	
287  			tdesc = tdesc - tmp;
288  		}
289  	
290  		if (ld_size) {
291  			mb[0] = rte_pktmbuf_alloc(mp);
292  			if (mb[0] == NULL) {
293  					throw std::runtime_error(" #####Cannot allocate mbuf packet");
294  			}
295  			memcpy(rte_pktmbuf_mtod(mb[0], void *), buffer, ld_size);
296  			buffer = static_cast<char*>(buffer) + ld_size;
297  			mb[0]->nb_segs = 1;
298  			mb[0]->next = NULL;
299  			rte_pktmbuf_data_len(mb[0]) = (uint16_t)ld_size;
300  			rte_pktmbuf_pkt_len(mb[0])	= (uint16_t)ld_size;
301  			nb_tx = rte_eth_tx_burst(portid, queueid, mb, 1);
302  			if(nb_tx == 0)
303  				rte_pktmbuf_free(mb[0]);
304  		}
305  		//printf("\nDMA transmitted number of packets: %d, on Queue-id:%d\n", nb_pkts, queueid);
306  		return 0;
307  	}

309  	queue_ext::queue_ext(instance& drv, uint32_t dev_id, uint32_t q_id, queue::mode mode,
310  				 queue::direction dir, const queue_ext::config& qcfg){
311  	
312  		m_drv = &drv;
313  		m_dev_id = dev_id;
314  		m_qid = q_id;
315  		m_mode = mode;
316  		m_dir = dir;
317  		m_buff_sz = qcfg.idx_bufsz;
318  		m_burst_size = qcfg.burst_size;
319  		char	mem_pool[RTE_MEMPOOL_NAMESIZE];
320  		uint32_t		nb_buff;
321  		struct rte_mempool	*mbuf_pool;
322  	
323  		/** Mbuf packet pool **/
324  		nb_buff = ((qcfg.idx_rngsz));
325  	
326  		nb_buff += NUM_RX_PKTS;
327  	
328  		nb_buff = RTE_MAX(nb_buff, (RTE_MEMPOOL_CACHE_MAX_SIZE * 1.5));
329  		if (dir == qdma::queue_ext::direction::both){
330  			snprintf(mem_pool, RTE_MEMPOOL_NAMESIZE, MBUF_POOL_NAME_PORT_QUEUE_DIR, m_dev_id, m_qid, (int)queue_ext::direction::host_to_card);
331  			mbuf_pool = rte_pktmbuf_pool_create(mem_pool, nb_buff, MP_CACHE_SZ,
332  					0, qcfg.idx_bufsz + RTE_PKTMBUF_HEADROOM, rte_socket_id());
333  			if (mbuf_pool == NULL)
334  				rte_exit(EXIT_FAILURE, " Cannot create mbuf pkt-pool for h2c-qid: %d, error: %s\n", m_qid, rte_strerror(rte_errno));
335  			snprintf(mem_pool, RTE_MEMPOOL_NAMESIZE, MBUF_POOL_NAME_PORT_QUEUE_DIR, m_dev_id, m_qid, (int)queue_ext::direction::card_to_host);
336  			mbuf_pool = rte_pktmbuf_pool_create(mem_pool, nb_buff, MP_CACHE_SZ,
337  					0, qcfg.idx_bufsz + RTE_PKTMBUF_HEADROOM, rte_socket_id());
338  	
339  			if (mbuf_pool == NULL)
340  				rte_exit(EXIT_FAILURE, " Cannot create mbuf pkt-pool for c2h-qid: %d, error: %s\n", m_qid, rte_strerror(rte_errno));
341  			m_drv->add_queue(m_dev_id, m_qid, 1, mode, qdma::queue_ext::direction::host_to_card, qcfg);
342  			m_drv->add_queue(m_dev_id, m_qid, 1, mode, qdma::queue_ext::direction::card_to_host, qcfg);
343  			m_drv->start_queue(m_dev_id, m_qid, mode, qdma::queue_ext::direction::host_to_card, qcfg);
344  			m_drv->start_queue(m_dev_id, m_qid, mode, qdma::queue_ext::direction::card_to_host, qcfg);
345  		}
346  		else{
347  			snprintf(mem_pool, RTE_MEMPOOL_NAMESIZE, MBUF_POOL_NAME_PORT_QUEUE_DIR, m_dev_id, m_qid, (int)m_dir);
348  			mbuf_pool = rte_pktmbuf_pool_create(mem_pool, nb_buff, MP_CACHE_SZ,
349  					0, qcfg.idx_bufsz + RTE_PKTMBUF_HEADROOM, rte_socket_id());
350  	
351  			if (mbuf_pool == NULL)
352  				rte_exit(EXIT_FAILURE, " Cannot create mbuf pkt-pool for qid: %d,  direction: %d, error: %s\n", m_qid, (int)m_dir, rte_strerror(rte_errno));
353  			m_drv->add_queue(m_dev_id, m_qid, 1, mode, dir, qcfg);
354  			m_drv->start_queue(m_dev_id, m_qid, mode, dir, qcfg);
355  		}
356  	}

421  	size_t queue_ext::write(void* buffer, const size_t size, const long device_offset, bool throw_err) {
422  	
423  		unsigned tot_num_desc = 1, ld_size = 0;
424  	
425  		(void)throw_err;
426  		if(size != 0)
427  			tot_num_desc = (size/ m_buff_sz) + ((size % m_buff_sz) ? 1 : 0);
428  		ld_size = size % m_buff_sz;
429  	
430  		if (m_mode == queue_ext::mode::memory_mapped)
431  			rte_pmd_qdma_set_mm_endpoint_addr(m_dev_id, m_qid, RTE_PMD_QDMA_TX, device_offset);
432  	
433  		return do_xmit(m_dev_id, buffer, m_qid, ld_size, tot_num_desc, (size == 0) ? 1 : 0, m_buff_sz, m_burst_size);
434  	}
435  	
436  	size_t queue_ext::read(void* buffer, const size_t size, const long device_offset, bool throw_err) {
437  	
438  		unsigned mm_tdesc, mm_ld_size = 0;
439  	
440  		(void)throw_err;
441  		if (size >= m_buff_sz) {
442  	
443  			if (size % m_buff_sz) {
444  				mm_tdesc = (size / m_buff_sz) + 1;
445  				mm_ld_size = size % m_buff_sz;
446  			}
447  			else
448  				mm_tdesc = (size / m_buff_sz);
449  		}
450  		else {
451  			mm_tdesc = 1;
452  			mm_ld_size = size % m_buff_sz;
453  		}
454  	
455  		if (m_mode == queue_ext::mode::streaming){
456  			return do_recv_st(m_dev_id, buffer, m_qid, size);
457  		}
458  		else {
459  	
460  			if (m_mode == queue_ext::mode::memory_mapped)
461  				rte_pmd_qdma_set_mm_endpoint_addr(m_dev_id, m_qid, RTE_PMD_QDMA_RX, device_offset);
462  	
463  			return do_recv_mm(m_dev_id, buffer, m_qid, mm_ld_size, mm_tdesc, m_buff_sz, m_burst_size);
464  		}
465  	
466  	}
