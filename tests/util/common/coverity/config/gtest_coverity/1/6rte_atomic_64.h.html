54   	static inline int
55   	rte_atomic64_cmpset(volatile uint64_t *dst, uint64_t exp, uint64_t src)
56   	{
57   		uint8_t res;
58   	
59   	
60   		asm volatile(
61   				MPLOCKED
62   				"cmpxchgq %[src], %[dst];"
63   				"sete %[res];"
64   				: [res] "=a" (res),     /* output */
65   				  [dst] "=m" (*dst)
66   				: [src] "r" (src),      /* input */
67   				  "a" (exp),
68   				  "m" (*dst)
69   				: "memory");            /* no-clobber list */
70   	
71   		return res;
72   	}

74   	static inline uint64_t
75   	rte_atomic64_exchange(volatile uint64_t *dst, uint64_t val)
76   	{
77   		asm volatile(
78   				MPLOCKED
79   				"xchgq %0, %1;"
80   				: "=r" (val), "=m" (*dst)
81   				: "0" (val),  "m" (*dst)
82   				: "memory");         /* no-clobber list */
83   		return val;
84   	}
