16   	device::device(instance& drv, const info& details)
17   	    : m_id(details.id), max_num_queues(details.max_num_queues), dev_cap(details.dev_cap), m_drv(drv) {
18   	
19   	
20   	    // save pointer to dma control bar
21   	    m_pcie_bars.reserve(3);
22   	
23   	    m_pcie_bars.emplace_back(m_id, details.dma_bar_idx);
24   	    dma_bar = m_pcie_bars.data();
25   	
26   	    // save pointer to user_bar
27   	    if (details.user_bar_idx != -1) {
28   	
29   	        m_pcie_bars.emplace_back(m_id, details.user_bar_idx);
30   	        user_bar = m_pcie_bars.data() + 1;
31   	    }
32   		 // save pointer to bypass_bar
33   		if (details.bypass_bar_idx != -1) {
34   	
35   			m_pcie_bars.emplace_back(m_id, details.bypass_bar_idx);
36   			bypass_bar = m_pcie_bars.data() + 2;
37   	    }
38   	}
