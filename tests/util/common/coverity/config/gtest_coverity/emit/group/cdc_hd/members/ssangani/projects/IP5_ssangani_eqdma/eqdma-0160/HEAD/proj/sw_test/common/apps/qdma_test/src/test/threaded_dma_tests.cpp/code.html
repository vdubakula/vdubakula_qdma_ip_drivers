97   	public:
98   	    std::vector<std::thread> workers;
99   	
100  	    template<typename Callable>
101  	    void execute_in_parallel(Callable& work, const threaded_test_params& tp) {
102  	        workers.reserve(tp.transfers.size());
103  	        for (const auto& tx : tp.transfers) {
104  	            workers.emplace_back(work, std::ref(tx));
105  	        }
106  	        for (auto& w : workers) {
107  	            w.join();
108  	        }
109  	    }
110  	};

117  	/// create test scenario where for each queue in num_q, it cycles through num_pf and executes num_it
118  	/// transfers of tx_size bytes.
119  	/// e.g.  interleaved_tx_test(2, 4, 1, 4K) produces:
120  	///     [0] = { pf=0, q=0, it=1, tx_size=4k }
121  	///     [1] = { pf=1, q=1, it=1, tx_size=4k }
122  	///     [2] = { pf=0, q=2, it=1, tx_size=4k }
123  	///     [3] = { pf=1, q=3, it=1, tx_size=4k }
124  	static threaded_test_params interleaved_tx_test(unsigned num_func, unsigned num_q, unsigned num_it,
125  	                                                size_t tx_size, qdma::instance_ext::type dev_type=qdma::instance_ext::type::pf) {
126  	
127  	    std::vector<threaded_test_params::transfer> transfers;
128  	    transfers.reserve(num_q);
129  	
130  	    std::vector<qdma::device> &devlist = env.get_dev_list(dev_type);
131  	
132  	    if (num_func > devlist.size())
133  			num_func = (unsigned int)devlist.size();
134  	
135  	#ifdef DPDK
136  	    std::vector<threaded_test_params::fmap> fmaps;
137  	    uint32_t base = 0;
138  	    fmaps.reserve(num_func);
139  	#endif
140  	
141  	    for (unsigned func_index = 0; func_index < num_func; func_index++) {
142  	    	if (!devlist[func_index].max_num_queues)
143  				continue;
144  	
145  	        if (num_q > devlist[func_index].max_num_queues) {
146  	        	num_q = devlist[func_index].max_num_queues;
147  	        }
148  	#ifdef DPDK
149  		    	fmaps.emplace_back(devlist[func_index].dev_id, base, num_q);
150  		    	base += num_q;
151  	#endif
152  			for (unsigned q = 0; q < num_q; q++) {
153  					transfers.emplace_back(devlist[func_index].dev_id, q, num_it, (uint32_t)tx_size, devlist[func_index]);
154  	        }
155  	    }
156  	#ifdef DPDK
157  	    return threaded_test_params { std::move(fmaps), std::move(transfers), dev_type};
158  	#else
159  	    return threaded_test_params { std::move(transfers), dev_type};
160  	#endif
161  	}

480  	            user_logic.set_st_loopback(false);
481  	#ifdef DPDK
482  	        	const auto hw_qid = get_hw_qid(tx.pf, tx.q);
483  	        	while(tx.flag)
484  	        		std::this_thread::sleep_for(std::chrono::milliseconds(1));
485  	        	user_logic.configure_c2h(hw_qid, tx.size, 1);
486  	        	user_logic.generate_packets();
487  	        	tx.flag = 1;
488  	        	std::this_thread::sleep_for(std::chrono::milliseconds(1)); // fixme
489  	#else
490  	        	const auto hw_qid = get_hw_qid(tx.pf, tx.q, devlist);
491  	        	user_logic.configure_c2h(hw_qid, tx.size, 1);
492  	        	user_logic.generate_packets();
493  	
494  	#ifndef _WIN32
495  	        	std::this_thread::sleep_for(std::chrono::milliseconds(1)); // fixme
496  	#endif
497  	
498  	#endif
499  	        }
500  	    }
501  	
502  	    // wait until threads complete
503  	    for (auto& w : workers) {
504  	        w.join();
505  	    }
506  	
507  	    // compare results
508  	    for (size_t i = 0; i < rd_buffers.size(); i++) {
509  	        auto cmp_result = std::memcmp(expected.data(), rd_buffers[i].data(),
510  	                tp.transfers[i].size * tp.transfers[i].it);
511  	        EXPECT_EQ(0, cmp_result) << "packet data of tx" << i << " does not match expected pattern!\n"
512  	                << print_diffs(expected.cbegin(), expected.cend(), rd_buffers[i].cbegin());
513  	    }
514  	#ifdef DPDK
515  	    c2h_qs.erase(c2h_qs.begin(), c2h_qs.end());
516  	    for(const auto& pf : tp.pfs){
517  	    	if (pf.qmax){
518  	    		//env.pfs[pf.dev_id].max_num_queues = 512;
519  	    		rte_eth_dev_close(pf.dev_id);
520  	    	}
521  	    }
522  	#endif
523  	}
