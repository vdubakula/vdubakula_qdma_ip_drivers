23   	std::vector<device> instance_ext::get_devices() {
24   		auto num_devs = get_num_devices();
25   	
26   		std::vector<device> qdma_devices;
27   		qdma_devices.reserve(num_devs);
28   		for (auto i = 0u; i < num_devs; i++) {
29   			const auto dev_details = dev_info(i);
30   			qdma_devices.emplace_back(*this, dev_details);
31   			qdma_devices[i].dev_id = dev_details.id;
32   		}
33   		return std::move(qdma_devices);
34   	}

38   	unsigned instance_ext::get_num_devices() {
39   	
40   		int num_ports, pf_ports = 0, vf_ports = 0;
41   		enum rte_pmd_qdma_pci_func_type func_type;
42   		num_ports = rte_eth_dev_count_avail();
43   		if (num_ports < 1)
44   			throw std::runtime_error("No Ethernet devices found Try updating the FPGA image");
45   		for(int i=0; i < num_ports; i++ ){
46   			rte_pmd_qdma_get_pci_func_type(i, &func_type);
47   			if (func_type == RTE_PMD_QDMA_PCI_FUNC_VF)
48   				vf_ports++;
49   			else
50   				pf_ports++;
51   		}
52   		if (m_type == instance_ext::type::pf )
53   			return pf_ports;
54   		else
55   			return vf_ports;
56   	}

58   	auto instance_ext::dev_info(unsigned dev_id) -> device::info {
59   	
60   		device::info info;
61   		int pf_ports = 0, vf_ports = 0, num_ports;
62   		int ret = 0;
63   		int *pf, *vf;
64   		struct rte_pmd_qdma_dev_attributes dev_attr;
65   	
66   		num_ports = rte_eth_dev_count_avail();
67   	
68   		/*Create port to pf,vf mapping.*/
69   		pf= new int[num_ports];
70   		vf= new int[num_ports];
71   	
72   		if (pf == nullptr || vf == nullptr)
73   			throw std::runtime_error("PF/VF memory allocation failed");
74   	
75   		for (int i =0; i < num_ports; i++) {
76   			/* Mark invalid*/
77   			pf[i] = -1;
78   			vf[i] = -1;
79   	
80   			enum rte_pmd_qdma_pci_func_type func_type;
81   			rte_pmd_qdma_get_pci_func_type(i, &func_type);
82   	
83   			if (func_type == RTE_PMD_QDMA_PCI_FUNC_VF) {
84   				vf[vf_ports] = i;
85   				vf_ports++;
86   			}
87   			if (func_type == RTE_PMD_QDMA_PCI_FUNC_PF) {
88   				pf[pf_ports] = i;
89   				pf_ports++;
90   			}
91   		}
92   	
93   		/* Get portid for a given pf/vf */
94   		if (m_type == instance_ext::type::vf )
95   			info.id = vf[dev_id];
96   		if (m_type == instance_ext::type::pf )
97   			info.id = pf[dev_id];
98   		delete[] pf;
99   		delete[] vf;
100  	
101  		ret = rte_pmd_qdma_get_bar_details(info.id, &(info.dma_bar_idx),
102  				&(info.user_bar_idx), &(info.bypass_bar_idx));
103  		if (ret < 0)
104  			throw std::runtime_error("rte_pmd_qdma_get_bar_details failed");
105  	
106  		ret = rte_pmd_qdma_get_device_capabilities(info.id, &dev_attr);
107  		if (ret < 0)
108  			throw std::runtime_error("rte_pmd_qdma_get_device_capabilities failed");
109  	
110  		info.dev_cap.num_pfs = dev_attr.num_pfs;
111  		info.dev_cap.num_qs = dev_attr.num_qs;
112  		info.dev_cap.flr_present = dev_attr.flr_present;
113  		info.dev_cap.st_en = dev_attr.st_en;
114  		info.dev_cap.mm_en = dev_attr.mm_en;
115  		info.dev_cap.mm_cmpt_en = dev_attr.mm_cmpt_en;
116  		info.dev_cap.mailbox_en = dev_attr.mailbox_en;
117  		info.dev_cap.mm_channel_max = dev_attr.mm_channel_max;
118  		info.dev_cap.cmpt_ovf_chk_dis = dev_attr.cmpt_ovf_chk_dis;
119  		info.dev_cap.sw_desc_64b = dev_attr.sw_desc_64b;
120  		info.dev_cap.cmpt_desc_64b = dev_attr.cmpt_desc_64b;
121  		info.dev_cap.cmpt_trig_count_timer = dev_attr.cmpt_trig_count_timer;
122  		info.dev_cap.device_type = dev_attr.device_type;
123  		info.dev_cap.versal_ip_type = dev_attr.versal_ip_type;
124  	
125  		return info;
126  	}

129  	unsigned instance_ext::add_queue(unsigned dev_id, unsigned q_id, __rte_unused unsigned num_q, queue_ext::mode q_mode,
130  								 queue_ext::direction q_dir, const queue_ext::config& qcfg, bool throw_err) {
131  	
132  		struct rte_mempool			*mbuf_pool;
133  		struct rte_eth_txconf		tx_conf;
134  		struct rte_eth_rxconf		rx_conf;
135  		int							ret;
136  		enum rte_pmd_qdma_rx_bypass_mode	rx_bypass_mode = RTE_PMD_QDMA_RX_BYPASS_NONE;
137  		char	mem_pool[RTE_MEMPOOL_NAMESIZE];
138  		enum rte_pmd_qdma_queue_mode_t queue_mode;
139  		struct rte_pmd_qdma_dev_attributes dev_attr;
140  	
141  		if(q_mode == queue_ext::mode::streaming) {
142  			queue_mode = RTE_PMD_QDMA_STREAMING_MODE;
143  		}
144  		else if(q_mode == queue_ext::mode::memory_mapped) {
145  			queue_mode = RTE_PMD_QDMA_MEMORY_MAPPED_MODE;
146  		}
147  		else {
148  			printf("%s, %d, invalid queue mode is specified, queue mode = %d\n", __func__, __LINE__, (int)q_mode);
149  		}
150  	
151  		(void)throw_err;
152  		snprintf(mem_pool, RTE_MEMPOOL_NAMESIZE, MBUF_POOL_NAME_PORT_QUEUE_DIR, dev_id, q_id, (int)q_dir);
153  		if ((mbuf_pool = rte_mempool_lookup(mem_pool)) == NULL)
154  			throw std::runtime_error("add_queue: Could not find mempool with name:");
155  		memset(&tx_conf, 0x0, sizeof(struct rte_eth_txconf));
156  		memset(&rx_conf, 0x0, sizeof(struct rte_eth_rxconf));
157  	
158  		ret = rte_pmd_qdma_get_device_capabilities(dev_id, &dev_attr);
159  		if (ret < 0)
160  			throw std::runtime_error("rte_pmd_qdma_get_device_capabilities failed");
161  	
162  		ret = rte_pmd_qdma_set_queue_mode(dev_id, q_id, queue_mode);
163  		if(ret < 0)
164  			throw std::runtime_error("rte_pmd_qdma_set_queue_mode  failed");
165  	
166  		if (q_dir == queue_ext::direction::host_to_card){
167  			tx_conf.tx_deferred_start = 1;
168  			if (qcfg.desc_bypass && (qcfg.sw_desc_sz == 64)) {
169  				ret = rte_pmd_qdma_configure_tx_bypass(dev_id, q_id,
170  						(rte_pmd_qdma_tx_bypass_mode)((qcfg.desc_bypass)?1:0),
171  						(rte_pmd_qdma_bypass_desc_len)qcfg.sw_desc_sz);
172  	
173  				if(ret < 0) {
174  					throw std::runtime_error("rte_pmd_qdma_configure_tx_bypass failed");
175  				}
176  			}
177  			ret = rte_eth_tx_queue_setup(dev_id, q_id, qcfg.idx_rngsz, 0,
178  						&tx_conf);
179  			if (ret < 0) {
180  				printf("rte_eth_tx_queue_setup failed with error %d for port_id = %d queue id = %d\n", ret, dev_id, q_id);
181  				throw std::runtime_error("Cannot setup TX Queue");
182  			}
183  		}
184  		else if (q_dir == queue_ext::direction::card_to_host){
185  			rx_conf.rx_deferred_start = 1;
186  	
187  			if((qcfg.desc_bypass == 1) && (qcfg.pfetch_bypass == 1)) {
188  				rx_bypass_mode = RTE_PMD_QDMA_RX_BYPASS_SIMPLE;
189  			}
190  			else if(qcfg.desc_bypass == 1) {
191  				rx_bypass_mode = RTE_PMD_QDMA_RX_BYPASS_CACHE;
192  			}
193  			ret = rte_pmd_qdma_configure_rx_bypass(dev_id, q_id, rx_bypass_mode, (rte_pmd_qdma_bypass_desc_len)0);
194  			if(ret < 0)
195  				throw std::runtime_error("rte_pmd_qdma_configure_rx_bypass failed");
196  	
197  			if(q_mode == queue_ext::mode::streaming) {
198  				ret = rte_pmd_qdma_set_cmpt_descriptor_size(dev_id, q_id, (rte_pmd_qdma_cmpt_desc_len )qcfg.cmptsz);
199  				if(ret < 0)
200  					throw std::runtime_error("rte_pmd_qdma_set_cmpt_descriptor_size failed");
201  			}
202  	
203  			/* Update the threshold in rxconf */
204  			rx_conf.rx_thresh.wthresh = qcfg.idx_cntr;
205  			ret = rte_eth_rx_queue_setup(dev_id, q_id, qcfg.idx_rngsz, 0,
206  							&rx_conf, mbuf_pool);
207  			if (ret < 0) {
208  				printf("rte_eth_rx_queue_setup failed with error %d for port_id = %d, queue id = %d\n", ret, dev_id, q_id);
209  				throw std::runtime_error("Cannot setup port %d RX Queue 0");
210  			}
211  	
212  			if(q_mode == queue_ext::mode::streaming) {
213  				ret = rte_pmd_qdma_set_c2h_descriptor_prefetch(dev_id, q_id, qcfg.prefetch);
214  				if(ret < 0)
215  					throw std::runtime_error("rte_pmd_qdma_set_c2h_descriptor_prefetch failed");
216  				if(!((dev_attr.device_type == QDMA_DEVICE_VERSAL) && (dev_attr.versal_ip_type == QDMA_VERSAL_HARD_IP))) {
217  					ret = rte_pmd_qdma_set_cmpt_overflow_check(dev_id, q_id, qcfg.c2h_ovf_chk_dis);
218  					if(ret < 0)
219  						throw std::runtime_error("rte_pmd_qdma_set_cmpt_overflow_check failed");
220  				}
221  				ret = rte_pmd_qdma_set_cmpt_trigger_mode(dev_id, q_id, (rte_pmd_qdma_tigger_mode_t)qcfg.trigmode);
222  				if(ret < 0)
223  					throw std::runtime_error("rte_pmd_qdma_set_cmpt_trigger_mode failed");
224  				ret = rte_pmd_qdma_set_cmpt_timer(dev_id, q_id, qcfg.idx_tmr);
225  				if(ret < 0)
226  					throw std::runtime_error("rte_pmd_qdma_set_cmpt_timer failed");
227  				ret = rte_pmd_qdma_set_immediate_data_state(dev_id, q_id, qcfg.c2h_udd_en);
228  				if(ret < 0)
229  					throw std::runtime_error("rte_pmd_qdma_set_immediate_data_state failed\n");
230  			}
231  		}
232  		else
233  			throw std::runtime_error("Undefined Direction");
234  	
235  		//std::cout << resp_str << "\n";
236  		return 0; //FIXME resp.get_attrib<uint32_t>(XNL_ATTR_QIDX);
237  	}

241  	unsigned instance_ext::start_queue(unsigned dev_id, unsigned queue_id, __rte_unused queue_ext::mode q_mode,
242  								  queue_ext::direction q_dir, __rte_unused const queue_ext::config& qcfg, bool throw_err) {
243  	
244  		int ret;
245  		struct rte_eth_dev *dev;
246  	
247  		/* DPDK v18.11.0 expects rte_eth_dev_start() to be done before starting any queue,
248  		 * gtest tests are acting on per qeueue base, so we explicitly mark device as started.
249  		 */
250  		(void)throw_err;
251  		dev = &rte_eth_devices[dev_id];
252  		dev->data->dev_started = 1;
253  	
254  		if (q_dir == queue_ext::direction::host_to_card){
255  			ret = rte_eth_dev_tx_queue_start(dev_id, queue_id);
256  			if (ret < 0 )
257  				throw std::runtime_error("Failed to start H2C queue");
258  		}
259  		else if (q_dir == queue_ext::direction::card_to_host){
260  			ret = rte_eth_dev_rx_queue_start(dev_id, queue_id);
261  			if (ret < 0 )
262  				throw std::runtime_error("Failed to start C2H queue");
263  		}
264  		return 0;
265  	}

267  	unsigned instance_ext::stop_queue(unsigned dev_id, unsigned queue_id, queue_ext::direction q_dir, bool throw_err) {
268  	
269  		int ret;
270  		struct rte_eth_dev *dev;
271  	
272  		(void)throw_err;
273  		dev = &rte_eth_devices[dev_id];
274  	
275  		if (q_dir == queue_ext::direction::host_to_card){
276  				ret = rte_eth_dev_tx_queue_stop(dev_id, queue_id);
277  				if (ret < 0 )
278  					throw std::runtime_error("Failed to stop H2C queue");
279  			}
280  			else if (q_dir == queue_ext::direction::card_to_host){
281  				ret = rte_eth_dev_rx_queue_stop(dev_id, queue_id);
282  				if (ret < 0 )
283  					throw std::runtime_error("Failed to stop C2H queue");
284  			}
285  		dev->data->dev_started = 0;
286  		return 0;
287  	}
