395  	// Each PF maintains its qid's relative to itself.
396  	// Thus if there are 2 PFs and each has 32 queues assigned, then:
397  	// - PF0 sw_qid 0 = hw_qid 0
398  	// - PF0 sw_qid 31 = hw_qid 31
399  	// - PF1 sw_qid 0 = hw_qid 32
400  	// - PF1 sw_qid 31 = hw_qid 63
401  	#ifdef DPDK
402  	inline unsigned get_hw_qid(unsigned vf_id, unsigned sw_qid) {
403  	#else
404  	inline unsigned get_hw_qid(unsigned vf_id, unsigned sw_qid,
405  									std::vector<qdma::device>& vfs) {
406  	#endif
407  		// get base hw_qid for this pf
408  		uint32_t vf_hw_qid_base = 0;
409  	
410  	#ifdef DPDK
411  		rte_pmd_qdma_get_queue_base(vf_id, &vf_hw_qid_base);
412  		return vf_hw_qid_base + sw_qid;
413  	#else
414  	
415  		for(const auto& vf : vfs) {
416  			if (vf_id == vf.dev_id) {
417  				vf_hw_qid_base = vf.qbase_start;
418  				break;
419  			}
420  		}
421  	
422  	    return vf_hw_qid_base + sw_qid;
423  	#endif
424  	}

522  	inline unsigned get_hw_qid(std::vector<qdma::device>::const_iterator vf_it,
523  			unsigned sw_qid, std::vector<qdma::device>& pfs,
524  			std::vector<qdma::device>& vfs) {
525  	
526  		auto sum_max_num_queues = [](unsigned sum, const qdma::device& dev) {
527  			return sum + dev.max_num_queues;
528  		};
529  	
530  		// sum of VF queues only (excluding this VF)
531  		unsigned vf_q_sum = std::accumulate(vfs.cbegin(), vf_it, 0u, sum_max_num_queues);
532  		std::cout << "\t vf_q_sum=" << vf_q_sum << "\n";
533  	
534  		// find the PF corresponding to this VF - FIXME assumes that all VFs are loaded for a given PF
535  		unsigned tmp = 0;
536  		const auto pf_it = std::find_if(pfs.begin(), pfs.end(), [&](const qdma::device& pf) -> bool {
537  			if (vf_q_sum >= tmp && (vf_q_sum <  tmp + pf.max_num_queues_vf)) {
538  				return true;
539  			} else {
540  				tmp += pf.max_num_queues_vf;
541  				return false;
542  			}
543  		});
544  		if(pf_it == pfs.end()) {
545  			throw std::runtime_error("could not find PF for this VF");
546  		}
547  		std::cout << "\t pf_idx=" << std::distance(pfs.begin(), pf_it) << "\n";
548  	
549  		// sum of all PF queues (including this PF)
550  		unsigned pf_q_sum = std::accumulate(pfs.begin(), pf_it, pf_it->max_num_queues, sum_max_num_queues);
551  		std::cout << "\t pf_q_sum=" << pf_q_sum << "\n";
552  	
553  		return pf_q_sum + vf_q_sum + sw_qid;
554  	}
