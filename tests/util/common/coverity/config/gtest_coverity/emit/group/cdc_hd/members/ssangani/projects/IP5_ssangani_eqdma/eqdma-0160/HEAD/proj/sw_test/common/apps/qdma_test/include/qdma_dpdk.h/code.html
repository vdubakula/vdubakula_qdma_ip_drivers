167  		/** @num_pfs - Num of PFs*/

171  		/** @flr_present - FLR resent or not? */
172  		uint8_t flr_present:1;
173  		/** @st_en - ST mode supported or not? */
174  		uint8_t st_en:1;
175  		/** @mm_en - MM mode supported or not? */
176  		uint8_t mm_en:1;
177  		/** @mm_cmpt_en - MM with Completions supported or not? */
178  		uint8_t mm_cmpt_en:1;
179  		/** @mailbox_en - Mailbox supported or not? */
180  		uint8_t mailbox_en:1;
181  		/** @mm_channel_max - Num of MM channels */
182  		uint8_t mm_channel_max;
183  	
184  		/** @cmpt_ovf_chk_dis - To indicate support of overflow check disable in CMPT ring */
185  		uint8_t cmpt_ovf_chk_dis:1;
186  		/** @sw_desc_64b - To indicate support of 64 bytes C2H/H2C descriptor format */
187  		uint8_t sw_desc_64b:1;
188  		/** @cmpt_desc_64b - To indicate support of 64 bytes CMPT descriptor format */
189  		uint8_t cmpt_desc_64b:1;
190  		/** @cmpt_trig_count_timer - To indicate support of counter + timer trigger mode */
191  		uint8_t cmpt_trig_count_timer:1;
192  		/** @device_type - Device Type */
193  		enum rte_pmd_qdma_device_type device_type;
194  		/** @versal_ip_type - Versal IP Type */
195  		enum rte_pmd_qdma_versal_ip_type versal_ip_type;
196  	};
