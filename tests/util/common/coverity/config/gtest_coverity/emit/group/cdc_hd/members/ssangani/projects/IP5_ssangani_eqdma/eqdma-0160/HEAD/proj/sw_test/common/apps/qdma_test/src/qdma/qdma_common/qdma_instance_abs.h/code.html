26

28   	
29   	    virtual device::info dev_info(unsigned dev_id) = 0;
30   	
31   	
32   	    virtual unsigned start_queue(unsigned dev_id, unsigned queue_id, queue::mode q_mode, queue_ext::direction q_dir,

34   	
35   	    virtual unsigned stop_queue(unsigned dev_id, unsigned queue_id, queue::direction q_dir, bool throw_err = true) = 0;
36   	
37   	    virtual unsigned add_queue(unsigned dev_id, unsigned q_id, unsigned num_q, queue::mode q_mode,
38   	                       queue::direction q_dir, const queue_ext::config& qcfg, bool throw_err = true) = 0;
39   	
40   	
41   		type m_type = type::pf;
42   	protected:
43   		uint32_t m_last_id;
44   	};
45   	
46   	inline std::vector<uint32_t> get_num_qs_per_pf(std::vector<device>& qdevs) {
47   	
48   	    std::vector<uint32_t> num_qs_per_pf;
49   	    num_qs_per_pf.reserve(qdevs.size());
50   	    std::transform(qdevs.cbegin(), qdevs.cend(), std::back_inserter(num_qs_per_pf),
51   	            [](const device& qdev) -> uint32_t {
52   	                    return qdev.max_num_queues;
53   	    });
54   	    return num_qs_per_pf;
55   	}
