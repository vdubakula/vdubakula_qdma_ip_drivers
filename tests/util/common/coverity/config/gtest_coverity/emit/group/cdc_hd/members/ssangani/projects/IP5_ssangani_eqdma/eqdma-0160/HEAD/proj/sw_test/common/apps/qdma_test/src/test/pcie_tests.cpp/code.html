22   	bool config_bar_access(qdma::instance_ext::type dev_type=qdma::instance_ext::type::pf)
23   	{
24   	    std::vector<qdma::device> &devlist = env.get_dev_list(dev_type);
25   	
26   	    for(uint32_t func = 0; func < devlist.size(); func++) {
27   	        auto& qdev = devlist[func];
28   	        if (!qdev.max_num_queues)
29   	            continue;
30   	
31   	        if (qdev.dma_bar == nullptr)
32   	            return false;
33   	
34   	        auto& control_bar = *qdev.dma_bar;
35   	
36   	        static const uint32_t addr = 0x244; // scratch register PF0
37   	
38   	        // precondition
39   	        if (control_bar.read<uint32_t>(addr) != 0x0)
40   	            return false;
41   	
42   	        // set all bits
43   	        control_bar.write(addr, 0xffffffff);
44   	        if (control_bar.read<uint32_t>(addr) != 0xffffffff)
45   	            return false;
46   	
47   	        // arbitrary pattern
48   	        control_bar.write(addr, 0xdeadbeef);
49   	        if (control_bar.read<uint32_t>(addr) != 0xdeadbeef)
50   	            return false;
51   	
52   	        // restore to old value
53   	        control_bar.write(addr, 0x0);
54   	        if (control_bar.read<uint32_t>(addr) != 0x0)
55   	            return false;
56   	    }
57   	    return true;
58   	}

60   	bool user_bar_access(qdma::instance_ext::type dev_type=qdma::instance_ext::type::pf)
61   	{
62   	    std::vector<qdma::device> &devlist = env.get_dev_list(dev_type);
63   	
64   	    for(uint32_t func = 0; func < devlist.size(); func++) {
65   	        auto& qdev = devlist[func];
66   	        if (!qdev.max_num_queues)
67   	            continue;
68   	
69   	        if (qdev.user_bar == nullptr)
70   	            return false;
71   	
72   	        auto& user_bar = *qdev.user_bar;
73   	
74   	        std::array<uint8_t, 0x100> buffer = { { 0 } };
75   	        std::iota(begin(buffer), end(buffer), uint8_t(0));
76   	
77   	        user_bar.write(0x100, buffer.data(), buffer.size());
78   	        buffer.fill(0xff);
79   	        user_bar.read(0x100, buffer.data(), buffer.size());
80   	
81   	        for (unsigned i = 0; i < buffer.size(); ++i) {
82   	            if (buffer[i] != i)
83   	                return false;
84   	        }
85   	    }
86   	    return true;
87   	}

89   	bool bypass_bar_access(qdma::instance_ext::type dev_type=qdma::instance_ext::type::pf)
90   	{
91   	    std::vector<qdma::device> &devlist = env.get_dev_list(dev_type);
92   	
93   	    for(uint32_t func = 0; func < devlist.size(); func++) {
94   	        auto& qdev = devlist[func];
95   	        if (!qdev.max_num_queues)
96   	            continue;
97   	
98   	        if (qdev.bypass_bar == nullptr)
99   	            return false;

112  	                return false;

116  	}
