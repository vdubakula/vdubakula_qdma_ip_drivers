18   		NO_BYPASS_MODE,     /* No Bypass */
19   		CACHE_BYPASS_MODE,	/* Cache bypass*/
20   		SIMPLE_BYPASS_MODE,	/* Simple bypass */
21   		DESC_BYPASS_MODE,	/* desc bypass mode for MM */

25   	/// The user logic part of the example design for the QDMA IP implements varies features for testing
26   	/// and demonstration purposes.
27   	///     - A data pattern checker for ST H2C transfers
28   	///     - A data pattern generator for ST C2H transfers
29   	///     - Various control bits
30   	/// Access to the user logic is given via the user BAR. Register map can be found here:
31   	/// xinc/acnd/engineering/ip/pcie/PCIe%20Gen4/AXI4MM/DMA/Everest/QDMA%20Specs/QDMA_Docs/QDMA%20user%20programming.docx
32   	class example_design_ctrl {
33   	public:
34   	
35   	    static constexpr size_t bram_size = 512_kB;
36   	
37   	    explicit example_design_ctrl(qdma::device& qdev)
38   	        : user_bar(*qdev.user_bar) {
39   	    }
40   	
41   	    //---------- data pattern checker specific ----------
42   	
43   	    bool check_h2c(const uint32_t qid, bool is_zero_byte_xfer) {
44   	        const auto match_reg = user_bar.read<uint32_t>(0x10);
45   	        const auto match_qid = get_bits(match_reg, 4, 11); // bits 4-14 = qid
46   	        const bool zero_byte_match = is_zero_byte_xfer? is_bit_set(match_reg, 1) : true;
47   	        return (match_qid == qid) && zero_byte_match && check_h2c_match(match_reg); // bit 0 = match
48   	    }
49   	
50   	    bool check_h2c_match() {
51   	        const auto match_reg = user_bar.read<uint32_t>(0x10);
52   	        return check_h2c_match(match_reg);
53   	    }
54   	
55   	    bool check_h2c_match(uint32_t match_reg) {
56   	        return is_bit_set(match_reg, 0);
57   	    }
58   	
59   	    void reset_h2c() {
60   	        user_bar.write<uint32_t>(0xC, 0x1); // clear match for h2c transfer pattern checker
61   	    }
62   	
63   	    //---------- data pattern generator specific ----------
64   	
65   	    using pattern_t = uint16_t;
66   	
67   	    void set_queue(const uint32_t qid) {
68   	        user_bar.write<uint32_t>(0x0, qid);
69   	    }
70   	
71   	    void configure_c2h(const uint32_t qid, const uint32_t packet_size,
72   	            const uint32_t num_packets) {
73   	        set_queue(qid);
74   	        user_bar.write<uint32_t>(0x4, packet_size);
75   	        user_bar.write<uint32_t>(0x20, num_packets);
76   	    }
77   	
78   	    void generate_packets() {
79   	        uint32_t regval = user_bar.read<uint32_t>(0x8);
80   	
81   	        regval = 0x02;
82   	        user_bar.write<uint32_t>(0x8, regval);
83   	    }
84   	
85   	    void reset_pkt_ctrl() {
86   	        user_bar.write<uint32_t>(0x8, 0x00);
87   	    }
88   	
89   	    void set_wrb_desc_size(xnl_st_c2h_cmpt_desc_size wrb_sz) {
90   	        uint32_t wrbsz_cfg = 0x00;
91   	
92   	        switch (wrb_sz) {
93   	            default:
94   	
95   	            case XNL_ST_C2H_CMPT_DESC_SIZE_8B:
96   	                wrbsz_cfg = 0x00;
97   	                break;
98   	            case XNL_ST_C2H_CMPT_DESC_SIZE_16B:
99   	                wrbsz_cfg = 0x01;
100  	                break;
101  	            case XNL_ST_C2H_CMPT_DESC_SIZE_32B:
102  	                wrbsz_cfg = 0x02;
103  	                break;
104  	            case XNL_ST_C2H_CMPT_DESC_SIZE_64B:
105  	                wrbsz_cfg = 0x03;
106  	                break;
107  	        }
108  	
109  	        user_bar.write<uint32_t>(0x50, wrbsz_cfg);
110  	    }
111

113  	    //---------- other features ----------
114  	
115  	    void set_bypass_mode(qdma::queue::mode mode, qdma::queue::direction dir, bypass_mode bypass) {
116  	        uint32_t reg_val = 0x00;
117  	
118  	        if(mode == qdma::queue::mode::memory_mapped) {
119  	        	switch (dir) {
120  	        		case qdma::queue::direction::host_to_card:
121  	        		{
122  	    	        	if( bypass == DESC_BYPASS_MODE)
123  	    	        		reg_val = 0x1;
124  	    	        	else
125  	    	        		reg_val = 0x0;
126  	
127  	    	        	break;
128  	        		}
129  	        		case qdma::queue::direction::card_to_host:
130  	        		{
131  	    	        	if( bypass == DESC_BYPASS_MODE)
132  	    	        		reg_val = 0x2;
133  	    	        	else
134  	    	        		reg_val = 0x0;
135  	
136  	    	        	break;
137  	        		}
138  	        		case qdma::queue::direction::both:
139  	        		{
140  	    	        	if( bypass == DESC_BYPASS_MODE)
141  	    	        		reg_val = 0x3;
142  	    	        	else
143  	    	        		reg_val = 0x0;
144  	
145  	    	        	break;
146  	        		}
147  	        		default:
148  	        			break;
149  	        	}
150  	        }
151  	        else
152  	        {
153  	            switch(dir) {
154  	    			case qdma::queue::direction::host_to_card:
155  	    			{
156  	    				switch (bypass) {
157  	    					default:
158  	    					case NO_BYPASS_MODE:
159  	    						reg_val = 0x00;
160  	    						break;
161  	    					case CACHE_BYPASS_MODE:
162  	    						reg_val = 0x1;
163  	    						break;
164  	    					case SIMPLE_BYPASS_MODE:
165  	    						reg_val = 0x1;
166  	    						break;
167  	    				}
168  	    				break;
169  	    			}
170  	    			case qdma::queue::direction::card_to_host:
171  	    			{
172  	    				switch (bypass) {
173  	    					default:
174  	    					case NO_BYPASS_MODE:
175  	    						reg_val = 0x00;
176  	    						break;
177  	    					case CACHE_BYPASS_MODE:
178  	    						reg_val = 0x2;
179  	    						break;
180  	    					case SIMPLE_BYPASS_MODE:
181  	    						reg_val = 0x4;
182  	    						break;
183  	    				}
184  	    				break;
185  	    			}
186  	    			case qdma::queue::direction::both:
187  	    			{
188  	    				switch (bypass) {
189  	    					default:
190  	    					case NO_BYPASS_MODE:
191  	    						reg_val = 0x00;
192  	    						break;
193  	    					case CACHE_BYPASS_MODE:
194  	    						reg_val = 0x3;
195  	    						break;
196  	    					case SIMPLE_BYPASS_MODE:
197  	    						reg_val = 0x5;
198  	    						break;
199  	    				}
200  	    				break;
201  	    			}
202  	    			default:
203  	    				break;
204  	            }
205  	        }
206  	
207  	        user_bar.write<uint32_t>(0x90, reg_val);
208  	    }

210  	    void set_desc_bypass_imm_data_loopback(bool on) {
211  	    	user_bar.write<uint32_t>(0x90, on ? 0x6 : 0x0);
212  	    }
213  	
214  	    void set_st_loopback(bool on) {
215  	        uint32_t regval = user_bar.read<uint32_t>(0x8);
216  	
217  	        regval &= ~0x01;
218  	        regval |= (on ? 0x1 : 0x0);
219  	        user_bar.write<uint32_t>(0x8, regval);
220  	    }
221  	
222  	    void set_imm_data_en(bool imm_on, int device_type, int versal_ip_type) {
223  	         uint32_t regval = 0;
224  	
225  	         if ((device_type == QDMA_DEVICE_VERSAL) && (versal_ip_type == QDMA_VERSAL_HARD_IP))
226  				 regval = (imm_on ? 0x06 : 0x00);
227  	         else
228  				 regval = (imm_on ? 0x04 : 0x00);
229  	
230  	         user_bar.write<uint32_t>(0x8, regval);
231  	     }
232  	
233  	
234  	    void set_imm_data(int idx, int val) {
235  	       user_bar.write<uint32_t>(0x30 + (4*idx), val);
236  	    }

238  	    /// Initiates a soft reset of the QDMA IP
239  	    /// This means that any csr, fmap or queue programming needs to be redone (i.e. reload the driver)
240  	    void reset_ip() {
241  	        user_bar.write<uint32_t>(0xA0, 0x1);
242  	    }
243  	
244  	private:
245  	    pcie_bar& user_bar;
246  	};
