43   	struct pci_id {
44   	    unsigned bus = 0;
45   	    unsigned dev = 0;
46   	    unsigned fn = 0;
47   	};
48   	
49   	inline std::string to_string(pci_id id) {
50   	    // 01:02:3
51   	    // where bus_id=0x01, dev_id=0x02, fn_id=3
52   	    std::stringstream ss;
53   	    ss << std::hex << std::setfill('0');
54   	    ss << std::setw(2) << id.bus << ":";
55   	    ss << std::setw(2) << id.dev << ".";
56   	    ss << id.fn;
57   	    return ss.str();
58   	}

60   	class pcie_bar{
61   	public:
62   	
63   		pcie_bar (uint32_t id, size_t bar) {
64   			bar_num = bar;
65   			m_id = id;
66   			pci_id bdf;
67   			struct rte_pci_device *pci_dev;
68   			pci_dev = RTE_ETH_DEV_TO_PCI(&rte_eth_devices[m_id]);
69   			bdf.bus = pci_dev->addr.bus;
70   			bdf.dev = pci_dev->addr.devid;
71   			bdf.fn = pci_dev->addr.function;
72   	
73   			const auto dev_path = get_device_resource_path(bdf);
74   			const auto bar_path = get_bar_resource_path(bdf, bar);
75   			std::ifstream resource_map(dev_path);
76   	
77   			for (size_t i = 0; i <= bar; i++) {
78   				size_t start, end, flags;
79   				resource_map >> std::hex >> start >> end >> flags;
80   				if (i == bar) {
81   					mapped_size = end - start + 1;
82   				}
83   			}
84   	
85   			int fd = open(bar_path.c_str(), O_RDWR);
86   			if (fd < 0) {
87   				throw std::system_error(errno, std::system_category(),
88   						"Failed to open bar resource! " + bar_path);
89   			}
90   	
91   			base = mmap(nullptr, mapped_size, PROT_WRITE | PROT_READ, MAP_SHARED, fd, 0);
92   			close(fd);
93   	
94   			if (base == MAP_FAILED) {
95   				mmap_fallback = 1;
96   				bar_num = bar;
97   				m_id = id;
98   			}
99   		}
100  	
101  	    ~pcie_bar() {
102  	        if (base && !mmap_fallback) {
103  	            munmap(base, mapped_size);
104  	        }
105  	    }
106  	
107  	    pcie_bar (const pcie_bar& rhs) = delete;
108  	    pcie_bar& operator=(const pcie_bar& rhs) = delete;
109  	
110  	    // Define move constructor and move assignment
111  	    pcie_bar(pcie_bar && rhs)  {
112  	        *this = std::move(rhs);
113  	    }
114  	    pcie_bar& operator=(pcie_bar&& rhs) {
115  	        if (this != &rhs) {
116  	            base = rhs.base; //pilfer
117  	            rhs.base = nullptr; //reset
118  	        }
119  	        return *this;
120  	    };
121  	
122  		template<typename T = uint32_t>
123  	    void write(size_t address, T t) {
124  	    	qdma_pci_write_reg(&rte_eth_devices[m_id], bar_num, address, t);
125  	    }
126  	
127  		template<typename T = uint32_t>
128  	    T read(size_t address) {
129  	    	T reg_val = (T) qdma_pci_read_reg(&rte_eth_devices[m_id], bar_num, address);
130  	    	return reg_val;
131  	    }
132  	
133  	    template<typename T = uint32_t>
134  	    void write(size_t address, T* t, size_t array_size) {
135  	    		std::memcpy(get_aligned_addr<T>(address), t, array_size * sizeof(T));
136  	    }
137  	
138  	    template<typename T = uint32_t>
139  	    void read(size_t address, T* t, size_t array_size) {
140  	        	std::memcpy(t, get_aligned_addr<T>(address), array_size * sizeof(T));
141  	     }
142  	
143  	    static std::string get_device_resource_path(pci_id id) {
144  	        // /sys/bus/pci/devices/0000:01:02:3/resource
145  	        // where bus_id=0x01, dev_id=0x02, fn_id=3
146  	        std::stringstream ss;
147  	        ss << std::hex << std::setfill('0');
148  	        ss << "/sys/bus/pci/devices/0000:" << to_string(id) << "/resource";
149  	        return ss.str();
150  	    }
151  	
152  	    static std::string get_bar_resource_path(pci_id id, uint32_t bar) {
153  	    std::stringstream ss;
154  	    ss << get_device_resource_path(id) << bar;
155  	        return ss.str();
156  	    }
157  	
158  	protected:
159  	    void* base = nullptr;
160  	    size_t mapped_size = 0;
161  	    bool mmap_fallback = 0;
162  	    unsigned int bar_num = 0;
163  	    uint32_t m_id;
164  	
165  	    template<typename T = uint32_t>
166  	    T* get_aligned_addr(size_t address) {
167  	        return &static_cast<T*>(base)[address / sizeof(T)];
168  	    }
169  	
170  	
171  	};
