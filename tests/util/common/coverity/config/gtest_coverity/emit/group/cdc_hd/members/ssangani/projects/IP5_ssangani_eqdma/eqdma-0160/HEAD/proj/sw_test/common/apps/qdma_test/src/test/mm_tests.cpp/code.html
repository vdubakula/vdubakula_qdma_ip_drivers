397  	static test_params basic_dma_test(const std::vector<uint32_t> &num_qs_per_pf, qdma::instance_ext::type dev_type=qdma::instance_ext::type::pf) {
398  		std::vector<test_params::transfer> transfers;
399  		auto num_transfer = 0;
400  		std::vector<qdma::device> &devlist = env.get_dev_list(dev_type);
401  	
402  	#ifdef DPDK
403  		std::vector<test_params::fmap> fmaps;
404  		uint32_t base = 0;
405  	#endif
406  		for (uint32_t pf_id = 0; pf_id < num_qs_per_pf.size(); pf_id++) {
407  			if (!devlist[pf_id].max_num_queues) {
408  				continue;
409  			}
410  			if (devlist[pf_id].max_num_queues < num_qs_per_pf[pf_id])
411  				continue;
412  			for (uint32_t qid = 0; qid < num_qs_per_pf[pf_id]; qid++) {
413  				num_transfer++;
414  			}
415  		}
416  	
417  	#ifdef DPDK
418  		fmaps.reserve(num_qs_per_pf.size());
419  		for(uint32_t pf_id =0; pf_id < num_qs_per_pf.size(); pf_id++){
420  			if (!devlist[pf_id].max_num_queues)
421  				continue;
422  			fmaps.emplace_back(devlist[pf_id].dev_id, base, num_qs_per_pf[pf_id]);
423  			base += num_qs_per_pf[pf_id];
424  		}
425  	#endif
426  	
427  	transfers.reserve(num_transfer);
428  	for (uint32_t pf_id = 0; pf_id < num_qs_per_pf.size(); pf_id++) {
429  		if (!devlist[pf_id].max_num_queues) {
430  			continue;
431  		}
432  		if (devlist[pf_id].max_num_queues < num_qs_per_pf[pf_id])
433  			continue;
434  		for (uint32_t qid = 0; qid < num_qs_per_pf[pf_id]; qid++) {
435  			transfers.emplace_back(devlist[pf_id].dev_id, qid, 1, 0, 4_kB, devlist[pf_id]);
436  		}
437  	}
438  	
439  	#ifdef DPDK
440  	return test_params { std::move(fmaps), std::move(transfers), dev_type };
441  	#else
442  	return test_params { std::move(transfers) , dev_type};
443  	#endif
444  	}
