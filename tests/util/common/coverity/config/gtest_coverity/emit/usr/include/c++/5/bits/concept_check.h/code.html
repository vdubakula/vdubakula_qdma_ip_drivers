31   	#define _CONCEPT_CHECK_H 1
32   	
33   	#pragma GCC system_header

35   	#include <bits/c++config.h>
36   	
37   	// All places in libstdc++-v3 where these are used, or /might/ be used, or
38   	// don't need to be used, or perhaps /should/ be used, are commented with
39   	// "concept requirements" (and maybe some more text).  So grep like crazy
40   	// if you're looking for additional places to use these.
41   	
42   	// Concept-checking code is off by default unless users turn it on via
43   	// configure options or editing c++config.h.
44   	
45   	#ifndef _GLIBCXX_CONCEPT_CHECKS
46   	
47   	#define __glibcxx_function_requires(...)
48   	#define __glibcxx_class_requires(_a,_b)
49   	#define __glibcxx_class_requires2(_a,_b,_c)
50   	#define __glibcxx_class_requires3(_a,_b,_c,_d)
51   	#define __glibcxx_class_requires4(_a,_b,_c,_d,_e)
52   	
53   	#else // the checks are on
54   	
55   	#include <bits/boost_concept_check.h>
56   	
57   	// Note that the obvious and elegant approach of
58   	//
59   	//#define glibcxx_function_requires(C) debug::function_requires< debug::C >()
60   	//
61   	// won't work due to concept templates with more than one parameter, e.g.,
62   	// BinaryPredicateConcept.  The preprocessor tries to split things up on
63   	// the commas in the template argument list.  We can't use an inner pair of
64   	// parenthesis to hide the commas, because "debug::(Temp<Foo,Bar>)" isn't
65   	// a valid instantiation pattern.  Thus, we steal a feature from C99.
66   	
67   	#define __glibcxx_function_requires(...)                                 \
68   	            __gnu_cxx::__function_requires< __gnu_cxx::__VA_ARGS__ >();
69   	#define __glibcxx_class_requires(_a,_C)                                  \
70   	            _GLIBCXX_CLASS_REQUIRES(_a, __gnu_cxx, _C);
71   	#define __glibcxx_class_requires2(_a,_b,_C)                              \
72   	            _GLIBCXX_CLASS_REQUIRES2(_a, _b, __gnu_cxx, _C);
73   	#define __glibcxx_class_requires3(_a,_b,_c,_C)                           \
74   	            _GLIBCXX_CLASS_REQUIRES3(_a, _b, _c, __gnu_cxx, _C);
75   	#define __glibcxx_class_requires4(_a,_b,_c,_d,_C)                        \
76   	            _GLIBCXX_CLASS_REQUIRES4(_a, _b, _c, _d, __gnu_cxx, _C);
77   	
78   	#endif // enable/disable
79   	
80   	#endif // _GLIBCXX_CONCEPT_CHECK
