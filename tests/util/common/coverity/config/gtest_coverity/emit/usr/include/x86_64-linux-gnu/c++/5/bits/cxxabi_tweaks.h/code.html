40   	  // The generic ABI uses the first byte of a 64-bit guard variable.
41   	#define _GLIBCXX_GUARD_TEST(x) (*(char *) (x) != 0)
42   	#define _GLIBCXX_GUARD_SET(x) *(char *) (x) = 1
43   	#define _GLIBCXX_GUARD_BIT __guard_test_bit (0, 1)
44   	#define _GLIBCXX_GUARD_PENDING_BIT __guard_test_bit (1, 1)
45   	#define _GLIBCXX_GUARD_WAITING_BIT __guard_test_bit (2, 1)
46   	  __extension__ typedef int __guard __attribute__((mode (__DI__)));
47   	
48   	  // __cxa_vec_ctor has void return type.
49   	  typedef void __cxa_vec_ctor_return_type;
50   	#define _GLIBCXX_CXA_VEC_CTOR_RETURN(x) return
51   	  // Constructors and destructors do not return a value.
52   	  typedef void __cxa_cdtor_return_type;
53
