24   	#ifndef _POPCNTINTRIN_H_INCLUDED
25   	#define _POPCNTINTRIN_H_INCLUDED
26   	
27   	#ifndef __POPCNT__
28   	#pragma GCC push_options
29   	#pragma GCC target("popcnt")
30   	#define __DISABLE_POPCNT__
31   	#endif /* __POPCNT__ */
32   	
33   	/* Calculate a number of bits set to 1.  */
34   	extern __inline int __attribute__((__gnu_inline__, __always_inline__, __artificial__))
35   	_mm_popcnt_u32 (unsigned int __X)
36   	{
37   	  return __builtin_popcount (__X);
38   	}
39   	
40   	#ifdef __x86_64__
41   	extern __inline long long  __attribute__((__gnu_inline__, __always_inline__, __artificial__))
42   	_mm_popcnt_u64 (unsigned long long __X)
43   	{
44   	  return __builtin_popcountll (__X);
45   	}
46   	#endif
47   	
48   	#ifdef __DISABLE_POPCNT__
49   	#undef __DISABLE_POPCNT__
50   	#pragma GCC pop_options
51   	#endif  /* __DISABLE_POPCNT__ */
52   	
53   	#endif /* _POPCNTINTRIN_H_INCLUDED */
