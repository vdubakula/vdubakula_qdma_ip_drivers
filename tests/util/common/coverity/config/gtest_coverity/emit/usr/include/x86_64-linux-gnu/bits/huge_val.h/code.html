24   	/* IEEE positive infinity (-HUGE_VAL is negative infinity).  */
25   	
26   	#if __GNUC_PREREQ(3,3)
27   	# define HUGE_VAL	(__builtin_huge_val())
28   	#elif __GNUC_PREREQ(2,96)
29   	# define HUGE_VAL	(__extension__ 0x1.0p2047)
30   	#elif defined __GNUC__
31   	
32   	# define HUGE_VAL \
33   	  (__extension__							      \
34   	   ((union { unsigned __l __attribute__((__mode__(__DI__))); double __d; })   \
35   	    { __l: 0x7ff0000000000000ULL }).__d)
36   	
37   	#else /* not GCC */
38   	
39   	# include <endian.h>
40   	
41   	typedef union { unsigned char __c[8]; double __d; } __huge_val_t;
42   	
43   	# if __BYTE_ORDER == __BIG_ENDIAN
44   	#  define __HUGE_VAL_bytes	{ 0x7f, 0xf0, 0, 0, 0, 0, 0, 0 }
45   	# endif
46   	# if __BYTE_ORDER == __LITTLE_ENDIAN
47   	#  define __HUGE_VAL_bytes	{ 0, 0, 0, 0, 0, 0, 0xf0, 0x7f }
48   	# endif
49   	
50   	static __huge_val_t __huge_val = { __HUGE_VAL_bytes };
51   	# define HUGE_VAL	(__huge_val.__d)
52   	
53   	#endif	/* GCC.  */
