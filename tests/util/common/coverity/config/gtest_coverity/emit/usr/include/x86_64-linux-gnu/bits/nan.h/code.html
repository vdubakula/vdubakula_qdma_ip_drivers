24   	/* IEEE Not A Number.  */
25   	
26   	#if __GNUC_PREREQ(3,3)
27   	
28   	# define NAN	(__builtin_nanf (""))
29   	
30   	#elif defined __GNUC__
31   	
32   	# define NAN \
33   	  (__extension__							      \
34   	   ((union { unsigned __l __attribute__ ((__mode__ (__SI__))); float __d; })  \
35   	    { __l: 0x7fc00000UL }).__d)
36   	
37   	#else
38   	
39   	# include <endian.h>
40   	
41   	# if __BYTE_ORDER == __BIG_ENDIAN
42   	#  define __qnan_bytes		{ 0x7f, 0xc0, 0, 0 }
43   	# endif
44   	# if __BYTE_ORDER == __LITTLE_ENDIAN
45   	#  define __qnan_bytes		{ 0, 0, 0xc0, 0x7f }
46   	# endif
47   	
48   	static union { unsigned char __c[4]; float __d; } __qnan_union
49   	  __attribute__ ((__unused__)) = { __qnan_bytes };
50   	# define NAN	(__qnan_union.__d)
51   	
52   	#endif	/* GCC.  */
